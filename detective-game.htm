    <script>

        // 游戏核心逻辑

        const gameLogic = {

            // 真正的凶手

            realKiller: "business-partner",

           

            // 关键证据

            keyEvidence: [

                "林海死亡时手中紧握的一张撕碎的名片",

                "书房桌上未寄出的信件，内容涉及商业机密",

                "厨房水槽里的咖啡杯，有轻微中毒痕迹"

            ],

           

            // 分析玩家的推理

            analyzeDeduction: function(selectedEvidence, selectedSuspect, deduction) {

                let score = 0;

                let feedback = [];

               

                // 检查是否选择了真凶

                if (selectedSuspect === this.realKiller) {

                    score += 30;

                    feedback.push("<p class='correct'>✓ 你成功指认了真凶！</p>");

                } else {

                    feedback.push("<p class='incorrect'>✗ 你指认的嫌疑人有误</p>");

                }

               

                // 检查关键证据

                let foundKeyEvidence = 0;

                selectedEvidence.forEach(evidence => {

                    if (this.keyEvidence.includes(evidence)) {

                        foundKeyEvidence++;

                    }

                });

               

                if (foundKeyEvidence === this.keyEvidence.length) {

                    score += 30;

                    feedback.push("<p class='correct'>✓ 你找到了所有关键证据！</p>");

                } else if (foundKeyEvidence > 0) {

                    score += foundKeyEvidence * 10;

                    feedback.push(`<p class='clue'>△ 你找到了部分关键证据（${foundKeyEvidence}/${this.keyEvidence.length}）</p>`);

                } else {

                    feedback.push("<p class='incorrect'>✗ 你没有找到任何关键证据</p>");

                }

               

                // 分析推理过程

                const lowerDeduction = deduction.toLowerCase();

                if (lowerDeduction.includes("商业") && lowerDeduction.includes("毒药") && lowerDeduction.includes("名片")) {

                    score += 30;

                    feedback.push("<p class='correct'>✓ 你的推理逻辑清晰，准确连接了所有线索！</p>");

                } else if (lowerDeduction.includes("商业") || lowerDeduction.includes("毒药") || lowerDeduction.includes("名片")) {

                    score += 15;

                    feedback.push("<p class='clue'>△ 你的推理部分正确，但还不够完整</p>");

                } else {

                    feedback.push("<p class='incorrect'>✗ 你的推理与关键线索关联不大</p>");

                }

               

                // 检查是否提到动机

                if (lowerDeduction.includes("冲突") || lowerDeduction.includes("利益") || lowerDeduction.includes("背叛")) {

                    score += 10;

                    feedback.push("<p class='correct'>✓ 你正确分析了凶手的动机！</p>");

                }

               

                return { score, feedback };

            },

           

            // 获取案件真相

            getCaseTruth: function() {

                return `

                    <div class="card" style="margin-top: 20px; border-color: #4CAF50;">

                        <h3 class="correct">案件真相</h3>

                        <p>真正的凶手是 <strong>张强</strong>，林海的商业合作伙伴。</p>

                        <p>动机：张强发现林海在商业合作中背叛了他，将重要的商业机密泄露给竞争对手，导致他损失惨重。</p>

                        <p>犯罪过程：张强以讨论商业合作的名义来到林海家中，在咖啡中下毒。林海在饮用后不久便感到不适，意识到了真相。在挣扎中，他紧握了张强的名片，但被张强慌乱中撕碎并试图掩盖。</p>

                        <p>关键证据解析：</p>

                        <ul style="text-align: left;">

                            <li><strong>撕碎的名片</strong> - 上面残留了张强的部分信息，是指向他的直接物证。</li>

                            <li><strong>未寄出的信件</strong> - 证明了林海确实存在泄露商业机密的行为，揭示了杀人动机。</li>

                            <li><strong>咖啡杯中的毒药</strong> - 是张强作案时使用的工具，上面可能留有他的指纹。</li>

                        </ul>

                    </div>

                `;

            }

        };

       

        document.getElementById('submit-btn').addEventListener('click', function() {

            const selectedEvidence = Array.from(document.querySelectorAll('.evidence-checkbox:checked'))

                .map(checkbox => checkbox.nextElementSibling.textContent);

            const selectedSuspect = document.getElementById('suspect-select').value;

            const deduction = document.getElementById('deduction-input').value;

           

            let resultText = '';

           

            if (selectedEvidence.length === 0) {

                resultText = '<p class="incorrect">请选择至少一个证据！</p>';

            } else if (!selectedSuspect) {

                resultText = '<p class="incorrect">请选择一个嫌疑人！</p>';

            } else if (deduction.trim() === '') {

                resultText = '<p class="incorrect">请输入你的推理过程！</p>';

            } else {

                // 分析玩家的推理

                const analysis = gameLogic.analyzeDeduction(selectedEvidence, selectedSuspect, deduction);

               

                // 构建结果文本

                const suspectText = document.getElementById('suspect-select').options[document.getElementById('suspect-select').selectedIndex].text;

 

                resultText = `

                    <h4>你的推理总结：</h4>

                    <p><strong>选择的证据：</strong>${selectedEvidence.join('，')}</p>

                    <p><strong>指控的嫌疑人：</strong>${suspectText}</p>

                    <p><strong>你的推理：</strong></p>

                    <p style="text-align: left; text-indent: 2em; padding: 10px; background-color: rgba(13, 26, 47, 0.5); border-radius: 4px;">${deduction.replace(/\n/g, '<br>')}</p>

                   

                    <h4 style="margin-top: 25px;">分析结果：</h4>

                    ${analysis.feedback.join('')}

                   

                    <h4>最终得分：${analysis.score}/100</h4>

                   

                    ${analysis.score >= 80 ? '<p class="correct" style="font-size: 1.2em;">🎉 恭喜你！你成功破解了这个案件！</p>' :

                      analysis.score >= 50 ? '<p class="clue">🔍 不错的尝试！你已经接近真相了，再仔细分析一下线索吧。</p>' :

                      '<p class="incorrect">❌ 很遗憾，你的推理还有很多漏洞。真相往往隐藏在细节之中。</p>'}

                   

                    <h4 style="margin-top: 30px; border-top: 1px solid #d4af37; padding-top: 20px;">案件真相：</h4>

                    ${gameLogic.getCaseTruth()}

                `;

            }

           

            document.getElementById('result-text').innerHTML = resultText;

            document.getElementById('result-modal').style.display = 'flex';

        });

       

        document.getElementById('close-modal').addEventListener('click', function() {

            document.getElementById('result-modal').style.display = 'none';

        });

       

        // 点击模态框外部也可关闭

        window.addEventListener('click', function(event) {

            if (event.target === document.getElementById('result-modal')) {

                document.getElementById('result-modal').style.display = 'none';

            }

        });

 

        // 按ESC键关闭模态框

        document.addEventListener('keydown', function(event) {

            if (event.key === 'Escape' && document.getElementById('result-modal').style.display === 'flex') {

                document.getElementById('result-modal').style.display = 'none';

            }

        });

    </script>
